583 template <class FT, class ET>
584 template <class Trait>
585 CONSTEXPR14 typename Trait::value_t Unpacked<FT, ET>::pack_xfixed() const {
586  switch (type) {
587  case Type::Infinity:
588  return 0;
589  case Type::Zero:
590  return 0;
591  case Type::NaN:
592  return 0;
593  default:
594  break;
595  }
596  constexpr int intbits = Trait::totalbits - Trait::fraction_bits;
597  if (exponent >= intbits) {
598  return negativeSign ? std::numeric_limits<typename Trait::value_t>::lowest()
599  : std::numeric_limits<typename Trait::value_t>::max();
600  } else if (exponent < -Trait::fraction_bits) {
601  return 0;
602  } else {
603  using ST = typename Trait::value_t;
604  using UT = typename int_least_bits<sizeof(ST) * 8>::unsigned_type;
605  // fraction 1.xxxxx from left aligned over FT bytes to UT bytes still left
606  // aligned over Trait::totalbits
607  UT f = fraction_bit_extract<FT_bits, FT, Trait::totalbits, UT,
608  (FT_bits > Trait::totalbits),
609  FT_leftmost_bit>::pack(fraction);
610 
611  // add the 1 bit for the current exponent
612  // f[totalbits] -> 0[intbits-exponent+1] 1
613  // f[exponent+Trait::fraction_bits-1]
614  //
615  // extrema: e.g. for totalbits=32, whatever fraction
616  // - exponent==-Trait::fraction_bits ==> 1 | 0
617  // - exponent==intbits-1 ==> 0x8000000 | (F >> 1)
618  ST r = (ST(1) << (exponent + Trait::fraction_bits)) |
619  (ST)(f >> (intbits - exponent));
620  return negativeSign ? -r : r;
621  }
