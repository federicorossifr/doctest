#include <iostream>
#include "posit.h"
typedef posit::Posit<int32_t,32,2,uint32_t,posit::PositSpec::WithNan> real;
#define points 8 /* for 2^8 = 256 points */
#define N (1 << points) /* N-point FFT */
typedef struct{
real Re;
real Im;
}
complex;
#ifndef PI
#define PI real(3.14159265358979323846264338327950288)
#endif
void fft(complex * wave, int n, complex * tmp) {
if (n > 1) {
/* return if n =< 0 */
int k, m;
complex z, w, *vo, *ve;
ve = tmp;
vo = tmp + n / 2;
for (k = 0; k < n / 2; k++) {
ve[k] = wave[2 * k];
vo[k] = wave[2 * k + 1];
}
fft(ve, n / 2, wave); // FFT on even-indexed elements of wave[]
fft(vo, n / 2, wave); // FFT on odd-indexed elements of wave[]
for (m = 0; m < n / 2; m++) {
w.Re = (std::cos((real(2) * PI * real(m) / real(n))));
w.Im = -(std::sin((real(2) * PI * real(m) / real(n))));
z.Re = w.Re * vo[m].Re - w.Im * vo[m].Im; /* Re(w*vo[m]) */
z.Im = w.Re * vo[m].Im + w.Im * vo[m].Re; /* Im(w*vo[m]) */
wave[m].Re = ve[m].Re + z.Re;
wave[m].Im = ve[m].Im + z.Im;
wave[m + n / 2].Re = ve[m].Re - z.Re;
wave[m + n / 2].Im = ve[m].Im - z.Im;
}
}
return;
}
// Program entry point.
int main() {
complex wave[N], scratch[N];
int k;
/* Fill wave[] as the sum of two pure sine waves of known frequency */
for (k = 0; k < N; k++) {
wave[k].Re = real(0.125) * std::cos((real(2) * PI * real(k) / real(N))) + real(0.5) * std::cos((real(2) * PI * real(10*k) / real(N)));
wave[k].Im = real(0.125) * std::sin((real(2) * PI * real(k) / real(N))) + real(0.5) * std::cos((real(2) * PI * real(10*k) / real(N)));
}
/* Printing the generated signal (for reading it in Matlab)*/
std::cout << "wave = [\n";
for (k = 0; k < N; k++) {
std::cout << wave[k].Re << "+" << wave[k].Im <<"i" << std::endl;
}
std::cout << "]; \n\n";
std::cout << "Computing the FFT...\n";
fft(wave, N, scratch); // Perform FFT, wave will have the result.
std::cout << "Printing the module of the obtained FFT coefficients:\n\nabs_FFT = [\n";
for (k = 0; k < N; k++) {
std::cout << std::sqrt((wave[k].Re*wave[k].Re+wave[k].Im*wave[k].Im)) << std::endl;
}
std::cout << "];\n\nDone. Check whether the second module is 32 and the 11th is 128\n";
std::cout << "2th: " << std::sqrt((wave[1].Re*wave[1].Re+wave[1].Im*wave[1].Im)) << std::endl;
std::cout << "11th: " << std::sqrt((wave[10].Re*wave[10].Re+wave[10].Im*wave[10].Im)) << std::endl;
std::cout << "(all the others must be zero, or close to zero)\n\n";
return 0;
}
