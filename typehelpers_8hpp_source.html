377  static inline typename unsigned128 extradiv(unsigned128 dividend,
378  uint64_t divisor) {
379  // https://github.com/abseil/abseil-cpp/blob/master/absl/numeric/int128.cc
380  // from http://stackoverflow.com/questions/5386377/division-without-using
381  if (divisor == dividend)
382  return 1;
383  else if (dividend < divisor)
384  return 0;
385  else {
386  unsigned128 denominator = divisor;
387  unsigned128 quotient = 0;
388 
389  // Left aligns the MSB of the denominator and the dividend.
390  const int shift =
391  findbitleftmostC(dividend) - findbitleftmostC(denominator);
392  denominator = denominator << shift;
393 
394  // Uses shift-subtract algorithm to divide dividend by denominator. The
395  // remainder will be left in dividend.
396  for (int i = 0; i <= shift; ++i) {
397  quotient = quotient << 1;
398  if (!(dividend < denominator)) {
399  dividend = dividend - denominator;
400  quotient.lo |= 1; // quotient |= 1;
401  }
402  denominator = denominator >> 1;
403  }
404  return quotient;
405  }
406  }
