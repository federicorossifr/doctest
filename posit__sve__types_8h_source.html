5 svint16_t float_to_posit16(svfloat32_t opv,svbool_t pg) {
6  svint32_t res_;
7  svuint32_t f_int = svreinterpret_u32(opv);
8  //__epi_4xi32_print(f_int,gvl,true);
9  svbool_t f_sign = svcmpne(pg,svand_z(pg,f_int,0x80000000),0);
10  svuint32_t f_exp = svlsr_z(pg,svlsl_z(pg,f_int,1),24);
11  svint32_t f_exp_norm = svsub_z(pg,svreinterpret_s32(f_exp),127); // also k-value
12  svuint32_t f_frac = svlsr_z(pg,f_int,10);
13  f_frac = svand_z(pg,f_frac,0x1FFF);
14 
15  // Exponent absolute value (LOW 8 bit -> 0xFFFFFF80)
16  svbool_t f_exp_norm_sign = svcmpne(pg,svand_z(pg,f_exp_norm,0xFFFFFF80),0);
17  svint32_t k_value = svadd_m(f_exp_norm_sign,svnot_m(f_exp_norm,f_exp_norm_sign,f_exp_norm),1);
18 
19  // Adjust k-value x:-x+1
20  svint32_t adj_k_value = svadd_m(f_exp_norm_sign,svneg_m(k_value,f_exp_norm_sign,k_value),1);
21 
22 
23  // Low 4 bits of k_value (AND with ____1111)
24  svuint32_t k_value_low = svreinterpret_u32(svand_z(pg,k_value,0xF));
25 
26  // Build regime with arithmetic right shift on signed integer
27  //Prepare the content wit [31-15]=1 [15-0]=0 1111 1111 1111 1111 1000 0000 0000 0000
28  svint32_t regime_hold = svdup_s32(0xFFFF8000);
29  regime_hold = svasr_z(pg,regime_hold,k_value_low);
30 
31  svint32_t regime_pos = svasr_z(pg,regime_hold,1);
32  svint32_t regime = svand_m(f_exp_norm_sign,regime_pos,svnot_z(pg,regime_hold));
33  regime = svand_z(pg,regime,0x00007FFF);
34 
35  // Compute regime length from least 4 significant bits of adjusted k-value x:-x+1
36  svint32_t regime_length = svand_z(pg,adj_k_value,0xF);
37 
38  // Use regime length to shift the mantissa right
39  f_frac = svlsr_z(pg,f_frac,svreinterpret_u32(regime_length));
40 
41  // Pack everything using OR
42  res_ = svorr_z(pg,regime,svreinterpret_s32(f_frac));
43 
44  // Use sign mask to perform 2's complement of Posit
45  res_ = svneg_m(res_,f_sign,res_);
46 
47 
48  return svreinterpret_s16(res_);
49 }

52  svfloat32_t posit_to_float(svint16_t opv,svbool_t pg) {
53  svint32_t res;
54  unsigned short POSIT_HOLDER_MSB = (1 << 15), POSIT_SIGNBIT = (1 << 15),POSIT_EXTRA_BITS = 0;
55  svbool_t sign_mask = svcmpne(pg,svand_z(pg,opv,POSIT_SIGNBIT),0);
56  //svint16_t n_one_helper = __builtin_epi_vmv_v_x_4xi16(-1,gvl);
57  //svint16_t one_helper = __builtin_epi_vmv_v_x_4xi16(1,gvl);
58 
59  svint16_t pa = svneg_m(opv,sign_mask,opv);
60  svint16_t pars1 = svlsl_z(pg,pa,POSIT_EXTRA_BITS+1);
61 
62  svbool_t one_marker_mask = svcmpne(pg,svand_z(pg,pars1,POSIT_HOLDER_MSB),0);
63  svbool_t neg_one_marker_mask = svcmpeq(pg,svand_z(pg,pars1,POSIT_HOLDER_MSB),0);
64  svint16_t neg_with_mask = svnot_m(pars1,one_marker_mask,pars1);
65 
66  svint16_t index = svsub_z(pg,svdup_s16(15),svreinterpret_s16(svclz_z(pg,neg_with_mask)));
67 
68  svint16_t kvalue = svadd_m(one_marker_mask,index,-1);
69  kvalue = svneg_m(kvalue,neg_one_marker_mask,kvalue);
70 
71  svint16_t rs = svadd_z(pg,index,1);
72 
73  //svint16_t rsexp = __builtin_epi_vadd_4xi16(rs,vbroadcast16(0,gvl),gvl);
74 
75  svint16_t pars2 = svlsl_z(pg,pars1,svreinterpret_u16(rs));
76 
77  //svint16_t fraction_nbits = __builtin_epi_vsub_4xi16(vbroadcast16(15,gvl),rsexp,gvl);
78  svint16_t fraction = pars2;
79 
80  // Adjust float sign in MSB
81  res = svdup_s32_z(sign_mask,0x80000000);
82 
83  // Adjust float exponent in MSBs
84  svint32_t f_exp = svadd_z(pg,svreinterpret_s32(kvalue),0x7f);
85  f_exp = svlsl_z(pg,svand_z(pg,f_exp,0xFF),23);
86 
87  // Adjust fraction in 23 LSB mantissa bits
88  svint32_t f_frac = svlsl_z(pg,svreinterpret_s32(fraction),7);
89 
90  // Pack the float
91  res = svorr_z(pg,res,f_exp);
92  res = svorr_z(pg,res,f_frac);
93  return svreinterpret_f32(res);
94 }
