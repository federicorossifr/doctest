364 __epi_8xi8 float_to_posit8(__epi_8xf32 opv,unsigned long gvl) {
365  __epi_8xi32 res_;
366  __epi_8xi32 f_int = (__epi_8xi32)opv;
367  //__epi_8xi32_print(f_int,gvl,true);
368  __epi_8xi1 f_sign = __builtin_epi_vmsne_8xi32(__builtin_epi_vand_8xi32(f_int,vbroadcast8xi32(0x80000000,gvl),gvl),vbroadcast8xi32(0,gvl),gvl);
369  __epi_8xi32 f_exp = __builtin_epi_vsrl_8xi32(__builtin_epi_vsll_8xi32(f_int,vbroadcast8xi32(1,gvl),gvl),vbroadcast8xi32(24,gvl),gvl);
370  __epi_8xi32 f_exp_norm = __builtin_epi_vsub_8xi32(f_exp,vbroadcast8xi32(127,gvl),gvl); // also k-value
371  __epi_8xi32 f_frac = __builtin_epi_vsrl_8xi32(f_int,vbroadcast8xi32(10,gvl),gvl);
372  f_frac = __builtin_epi_vand_8xi32(f_frac,vbroadcast8xi32(0x1FFF,gvl),gvl);
373 
374  // Exponent absolute value (LOW 8 bit -> 0xFFFFFF80)
375  __epi_8xi1 f_exp_norm_sign = __builtin_epi_vmsne_8xi32(__builtin_epi_vand_8xi32(f_exp_norm,vbroadcast8xi32(0xFFFFFF80,gvl),gvl),vbroadcast8xi32(0,gvl),gvl);
376  __epi_8xi32 k_value = __builtin_epi_8xi32_abs(f_exp_norm,0xFFFFFF80,gvl);
377  //__epi_8xi32_print(k_value,gvl,true);
378 
379  // Adjust k-value x:-x+1
380  __epi_8xi32 adj_k_value =__builtin_epi_8xi32_adj_k(f_exp_norm,f_exp_norm_sign,gvl);
381  //__epi_8xi32_print(adj_k_value,gvl,true);
382 
383  // Low 4 bits of k_value (AND with ____1111)
384  __epi_8xi32 k_value_low = __builtin_epi_vand_8xi32(k_value,vbroadcast8xi32(0xF,gvl),gvl);
385 
386  // Build regime with arithmetic right shift on signed integer
387  //Prepare the content wit [31-15]=1 [15-0]=0 1111 1111 1111 1111 1000 0000 0000 0000
388  __epi_8xi32 regime_hold = vbroadcast8xi32(0xFFFF8000,gvl);
389  regime_hold = __builtin_epi_vsra_8xi32(regime_hold,k_value_low,gvl);
390  __epi_8xi32 regime_pos = __builtin_epi_vsra_8xi32(regime_hold,vbroadcast8xi32(1,gvl),gvl);
391  __epi_8xi32 regime = __builtin_epi_vand_8xi32_mask(regime_pos,regime_pos,__builtin_epi_vnot_8xi32(regime_hold,gvl),f_exp_norm_sign,gvl);
392  regime = __builtin_epi_vand_8xi32(regime,vbroadcast8xi32(0x00007FFF,gvl),gvl);
393  //__epi_8xi32_print(regime,gvl,true);
394 
395  // Compute regime length from least 4 significant bits of adjusted k-value x:-x+1
396  __epi_8xi32 regime_length = __builtin_epi_vand_8xi32(adj_k_value,vbroadcast8xi32(0xF,gvl),gvl);
397 
398  // Use regime length to shift the mantissa right
399  f_frac = __builtin_epi_vsrl_8xi32(f_frac,regime_length,gvl);
400 
401  // Pack everything using OR
402  res_ = __builtin_epi_vor_8xi32(regime,f_frac,gvl);
403  //__epi_8xi32_print(res_,gvl,true);
404 
405  // Use sign mask to perform 2's complement of Posit
406  res_ = __builtin_epi_8xi32_abs(res_,f_sign,gvl);
407  //__epi_8xi32_print(res_,gvl,true);
408 
409 
410  return vreduce32_8(res_,gvl);
411 }
